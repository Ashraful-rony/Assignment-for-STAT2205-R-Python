\section{Question 8: Weekend vs Weekday Price}

\subsection{R Code}

\begin{lstlisting}[language=R]
aggregate(new_coach_price ~ weekend,
          data = airline_random_1000,
          mean)

\subsection{Interpretation}

\begin{itemize}

\item T-test checks statistical significance.
\item p-value below 0.05 means significant difference.
\end{itemize}
